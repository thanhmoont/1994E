background-position:
    thuộc tính xác định vị trí của ảnh nền
    nó chỉ có tác dụng khi kích cỡ của đối tượng lớn hơn kích cỡ ảnh nền
    và giá trị của thuộc tính background-repeat: no-repeat;

    background-position đặt vị trí theo trục x chiều ngang
    trục y là chiều dọc.
    background-position: top left
